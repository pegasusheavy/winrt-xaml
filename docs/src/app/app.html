Quick Start

1. Add to Your Cargo.toml

[dependencies]
winrt-xaml = "0.1.0"
windows = {{ '{' }}version = "0.58", features = ["Win32_Foundation", "Win32_UI_WindowsAndMessaging"]{{ '}' }}

2. Build the C++ Helper (One-time)

cd xaml_islands_helper
./build.bat

3. Create Your First WinRT App

use winrt_xaml::xaml_native::*;
use windows::Win32::System::Com::{{'{'}CoInitializeEx, CoUninitialize, COINIT_APARTMENTTHREADED{'}'}}

;

fn main() -> Result<()> {{'{'}
    // Initialize COM
    unsafe {{'{'} CoInitializeEx(None, COINIT_APARTMENTTHREADED).ok(); {'}'}

    // Initialize XAML
    let _manager = XamlManager::new()?;

    // Create host window
    let host_hwnd = create_host_window()?;

    // Create XAML source and attach to window
    let mut xaml_source = XamlSource::new()?;
    let island_hwnd = xaml_source.attach_to_window(host_hwnd)?;

    // Create a button
    let button = XamlButton::new()?;
    button.set_content("Hello WinRT!")?;
    button.set_size(200.0, 50.0)?;

    // Display it
    xaml_source.set_content_element(&button.as_uielement())?;

    // Show and run
    unsafe {{'{'} ShowWindow(island_hwnd, SW_SHOW); {'}'}
    run_message_loop()?;

    unsafe {{'{'} CoUninitialize(); {'}'}
    Ok(())
{'}'}

API Reference

XamlManager

Manages the Windows XAML framework. Must be kept alive for the duration of XAML usage.

let _manager = XamlManager::new()?;

XamlSource

DesktopWindowXamlSource for hosting XAML content in Win32 windows.

let mut source = XamlSource::new()?;
let island_hwnd = source.attach_to_window(host_hwnd)?;
source.set_content_element(&element.as_uielement())?;

XamlButton

Interactive button with click events, styling, and modern design.

let button = XamlButton::new()?;
button.set_content("Click Me")?;
button.set_size(200.0, 50.0)?;
button.set_background(0xFF0078D4)?;  // Microsoft blue
button.set_corner_radius(10.0)?;
button.on_click(|| {{'{'}
    println!("Button clicked!");
{'}'}})?;

XamlTextBox

Text input/output control with text retrieval.

let textbox = XamlTextBox::new()?;
textbox.set_placeholder("Enter text...")?;
textbox.set_size(300.0, 56.0)?;
textbox.set_background(0xFF2D2D2D)?;

// Get text from textbox
let text = textbox.get_text()?;

XamlStackPanel

Layout container with vertical/horizontal orientation and spacing.

let panel = XamlStackPanel::new()?;
panel.set_vertical(true)?;
panel.set_spacing(20.0)?;
panel.set_background(0xFF1A1A1A)?;
panel.set_padding(30.0, 30.0, 30.0, 30.0)?;
panel.add_child(&button.as_uielement())?;

XamlScrollViewer

Scrollable container for content that exceeds window bounds.

let scroll = XamlScrollViewer::new()?;
scroll.set_vertical_scroll_mode(ScrollMode::Enabled)?;
scroll.set_vertical_scrollbar_visibility(ScrollBarVisibility::Auto)?;
scroll.set_content(&panel.as_uielement())?;
